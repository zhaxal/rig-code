# Python dependencies for the spatial detection touchscreen app.
#
# Install on Raspberry Pi OS 64-bit with ./setup.sh, or manually:
#   python3 -m venv venv
#   source venv/bin/activate
#   pip install -r requirements.txt
#
# NOTE: two non-pip system packages are also required (installed by setup.sh):
#   - python3-tk : provides the tkinter GUI toolkit
#   - ffmpeg     : optional, for re-muxing recordings if needed

depthai>=3.0.0      # DepthAI v3 - OAK device API + SpatialDetectionNetwork
numpy               # frame handling
pillow              # converts OpenCV frames to tkinter images (ImageTk)
# opencv-python MUST be last: depthai pulls in opencv-python-headless,
# so installing opencv-python after it ensures the full build wins.
opencv-python       # frame drawing (boxes / overlay) + image saving
